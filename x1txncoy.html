# ip6tables 是 IPv6 防火墙，没有IPv6 也可以不执行。
ip6tables -I INPUT -m state --state NEW -m tcp -p tcp --dport 端口 -j ACCEPT
ip6tables -I INPUT -m state --state NEW -m udp -p udp --dport 端口 -j ACCEPT

# 添加完毕后，保存防火墙规则。
# CentOS 系统：
service iptables save
service ip6tables save

# Debian/Ubuntu 系统：
iptables-save > /etc/iptables.up.rules
ip6tables-save > /etc/ip6tables.up.rules
